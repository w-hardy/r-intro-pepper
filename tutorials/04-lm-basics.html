5  DRAFT: Linear models: a minimal example

Fit and interpret a simple linear model.

library(dplyr) # Data manipulation
library(broom) # Statistical output
library(ggplot2) # Figures


fit <- lm(bill_len ~ bill_dep + species, data = penguins)
broom::tidy(fit)
# Visual check

penguins |> 
  ggplot(aes(x = bill_dep, y = bill_len, colour = species)) +
  geom_point() +
  geom_smooth(method = "lm", se = TRUE) +
  labs(title = "Linear fit") +
  theme_minimal()
